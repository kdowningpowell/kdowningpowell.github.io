    const starCanvas = document.getElementById("stars");

    const starCtx = starCanvas ? starCanvas.getContext("2d") : null;


    function resizeStars() {

      if (!starCanvas) return;

      starCanvas.width = window.innerWidth;

      starCanvas.height = window.innerHeight;

    }


    resizeStars();

    window.addEventListener("resize", resizeStars);


    const stars = Array.from({ length: 80 }, () => ({

      x: Math.random() * window.innerWidth,

      y: Math.random() * window.innerHeight,

      r: Math.random() * 1.5 + 0.2,

      dx: (Math.random() - 0.5) * 0.3,

      dy: (Math.random() - 0.5) * 0.3

    }));


    function animateStars() {

      if (!starCanvas || !starCtx) return;

      starCtx.clearRect(0, 0, starCanvas.width, starCanvas.height);

      starCtx.fillStyle = "rgba(148, 163, 184, 0.8)";


      stars.forEach((s) => {

        starCtx.beginPath();

        starCtx.arc(s.x, s.y, s.r, 0, Math.PI * 2);

        starCtx.fill();

        s.x += s.dx;

        s.y += s.dy;

        if (s.x < 0 || s.x > starCanvas.width) s.dx *= -1;

        if (s.y < 0 || s.y > starCanvas.height) s.dy *= -1;

      });


      for (let i = 0; i < stars.length; i++) {

        for (let j = i + 1; j < stars.length; j++) {

          const dx = stars[i].x - stars[j].x;

          const dy = stars[i].y - stars[j].y;

          const dist = Math.sqrt(dx * dx + dy * dy);

          if (dist < 120) {

            starCtx.strokeStyle = "rgba(56, 189, 248, 0.15)";

            starCtx.lineWidth = 1;

            starCtx.beginPath();

            starCtx.moveTo(stars[i].x, stars[i].y);

            starCtx.lineTo(stars[j].x, stars[j].y);

            starCtx.stroke();

          }

        }

      }

      requestAnimationFrame(animateStars);

    }

    animateStars();

    const battleshipBoard = document.getElementById("battleshipBoard");

    const gameStatus = document.getElementById("gameStatus");

    const resetBattleshipBtn = document.getElementById("resetBattleshipBtn");

    const BOARD_SIZE = 8;

    const SHIPS = [5, 4, 3, 3, 2];

    let shipGrid = [];

    let guessedCells = [];

    let remainingHits = 0;

    let gameFinished = false;


    function makeEmptyBoard() {

      return Array.from({ length: BOARD_SIZE }, () => Array(BOARD_SIZE).fill(0));

    }


    function canPlaceShip(length, row, col, horizontal) {

      if (horizontal && col + length > BOARD_SIZE) return false;

      if (!horizontal && row + length > BOARD_SIZE) return false;

      for (let i = 0; i < length; i++) {

        const targetRow = horizontal ? row : row + i;

        const targetCol = horizontal ? col + i : col;

        if (shipGrid[targetRow][targetCol] === 1) return false;

      }

      return true;

    }


    function placeShipsRandomly() {

      shipGrid = makeEmptyBoard();

      guessedCells = makeEmptyBoard();

      remainingHits = SHIPS.reduce((sum, len) => sum + len, 0);

      gameFinished = false;

      SHIPS.forEach((length) => {

        let placed = false;

        while (!placed) {

          const horizontal = Math.random() < 0.5;

          const row = Math.floor(Math.random() * BOARD_SIZE);

          const col = Math.floor(Math.random() * BOARD_SIZE);

          if (canPlaceShip(length, row, col, horizontal)) {

            for (let i = 0; i < length; i++) {

              const targetRow = horizontal ? row : row + i;

              const targetCol = horizontal ? col + i : col;

              shipGrid[targetRow][targetCol] = 1;

            }

            placed = true;

          }

        }

      });

    }


    function renderBattleshipBoard() {

      if (!battleshipBoard) return;

      battleshipBoard.innerHTML = "";

      for (let row = 0; row < BOARD_SIZE; row++) {

        for (let col = 0; col < BOARD_SIZE; col++) {

          const cell = document.createElement("button");

          cell.type = "button";

          cell.className = "battleship-cell";

          cell.setAttribute("aria-label", `Row ${row + 1}, Column ${col + 1}`);

          if (guessedCells[row][col] === 1) {

            cell.classList.add("hit");

            cell.textContent = "X";

            cell.disabled = true;

          } else if (guessedCells[row][col] === 2) {

            cell.classList.add("miss");

            cell.textContent = "•";

            cell.disabled = true;

          } else if (gameFinished) {

            cell.disabled = true;

          }

          cell.addEventListener("click", function () { handleGuess(row, col); });

          battleshipBoard.appendChild(cell);

        }

      }

    }


    function handleGuess(row, col) {

      if (gameFinished || guessedCells[row][col] !== 0) return;

      if (shipGrid[row][col] === 1) {

        guessedCells[row][col] = 1;

        remainingHits--;

        gameStatus.textContent = `Hit! ${remainingHits} ship spaces remaining.`;

      } else {

        guessedCells[row][col] = 2;

        gameStatus.textContent = "Miss! Keep searching the grid.";

      }

      if (remainingHits === 0) {

        gameFinished = true;

        gameStatus.textContent = "Congratulations! You sank the full fleet in Battleship.";

      }

      renderBattleshipBoard();

    }


    function startBattleshipGame() {

      placeShipsRandomly();

      if (gameStatus) gameStatus.textContent = "Game loaded. Start by clicking any square on the board.";

      renderBattleshipBoard();

    }


    if (resetBattleshipBtn) {

      resetBattleshipBtn.addEventListener("click", startBattleshipGame);

      startBattleshipGame();

    }

    function drawRadarChart() {

      const radarCanvas = document.getElementById("skillRadar");

      const radarCtx = radarCanvas ? radarCanvas.getContext("2d") : null;

      if (!radarCanvas || !radarCtx) return;


      const size = radarCanvas.width;

      const cx = size / 2;

      const cy = size / 2;

      const radius = 78;

      const labels = ["Web", "JS", "Python", "Java", "DevSecOps", "Writing", "Agile"];

      const values = [90, 85, 80, 78, 82, 88, 90];

      const levels = 5;


      radarCtx.clearRect(0, 0, radarCanvas.width, radarCanvas.height);

      radarCtx.strokeStyle = "rgba(148, 163, 184, 0.28)";

      radarCtx.lineWidth = 1;


      for (let level = 1; level <= levels; level++) {

        const r = (radius / levels) * level;

        radarCtx.beginPath();

        for (let i = 0; i < labels.length; i++) {

          const angle = (Math.PI * 2 / labels.length) * i - Math.PI / 2;

          const x = cx + Math.cos(angle) * r;

          const y = cy + Math.sin(angle) * r;

          if (i === 0) radarCtx.moveTo(x, y);

          else radarCtx.lineTo(x, y);

        }

        radarCtx.closePath();

        radarCtx.stroke();

      }


      for (let i = 0; i < labels.length; i++) {

        const angle = (Math.PI * 2 / labels.length) * i - Math.PI / 2;

        const x = cx + Math.cos(angle) * radius;

        const y = cy + Math.sin(angle) * radius;

        radarCtx.beginPath();

        radarCtx.moveTo(cx, cy);

        radarCtx.lineTo(x, y);

        radarCtx.stroke();

        radarCtx.fillStyle = getComputedStyle(document.body).getPropertyValue("--muted").trim();

        radarCtx.font = "11px Arial";

        radarCtx.textAlign = x >= cx ? "left" : "right";

        radarCtx.textBaseline = "middle";

        radarCtx.fillText(labels[i], cx + Math.cos(angle) * (radius + 16), cy + Math.sin(angle) * (radius + 16));

      }


      radarCtx.beginPath();

      values.forEach((value, i) => {

        const angle = (Math.PI * 2 / labels.length) * i - Math.PI / 2;

        const r = (value / 100) * radius;

        const x = cx + Math.cos(angle) * r;

        const y = cy + Math.sin(angle) * r;

        if (i === 0) radarCtx.moveTo(x, y);

        else radarCtx.lineTo(x, y);

      });

      radarCtx.closePath();

      radarCtx.fillStyle = "rgba(56, 189, 248, 0.22)";

      radarCtx.strokeStyle = "rgba(56, 189, 248, 0.95)";

      radarCtx.lineWidth = 2;

      radarCtx.fill();

      radarCtx.stroke();


      values.forEach((value, i) => {

        const angle = (Math.PI * 2 / labels.length) * i - Math.PI / 2;

        const r = (value / 100) * radius;

        const x = cx + Math.cos(angle) * r;

        const y = cy + Math.sin(angle) * r;

        radarCtx.beginPath();

        radarCtx.arc(x, y, 3, 0, Math.PI * 2);

        radarCtx.fillStyle = "rgba(56, 189, 248, 1)";

        radarCtx.fill();

      });

    }
